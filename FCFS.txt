FCFS
#include<iostream>
using namespace std;
int main()
{
	int n=3;
	int AT[]={0,0,0};
	int BT[]={24,3,3};
	int CT[n],TAT[n],WAT[n];
	CT[0]=BT[0];
	for(int i=1;i<n;i++){
		CT[i]=CT[i-1]+BT[i];
		
	}
	for(int i=0;i<n;i++){
		TAT[i]=CT[i]-AT[i];
		WAT[i]=TAT[i]-BT[i];
		
	}
	cout <<"process\tAT\tBT\tCT\tTAT\tWAT\n";
	for(int i=0;i<n;i++){
	cout <<"P"<<i+1<<"\t"<<AT[i]<<"\t"<<BT[i]<<"\t\t"<<CT[i]<<"\t\t"<<TAT[i]<<"\t\t"<<WAT[i]<<"\n";	
	}
	return 0;
}
